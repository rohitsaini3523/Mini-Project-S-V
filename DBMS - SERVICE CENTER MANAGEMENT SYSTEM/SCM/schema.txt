create database vehicle_service_center;
use vehicle_service_center;
create table customer (cust_id int primary key auto_increment, cust_name varchar(20), cust_address varchar(20), cust_phone varchar(20), cust_vehicle_no varchar(10));
create table vehicle(vehicle_number varchar(10) primary key, vehicle_type varchar(20), RCbook varchar(20));
create table parts(part_no int primary key, part_name varchar(20), part_cost int, par_manufacturedate DATE, part_warrantyperiod DATE);
create table employee(emp_id int primary key auto_increment, emp_name varchar(20), emp_address varchar(50), emp_phone varchar(20), emp_salary int,emp_emailid varchar(60),emp_password varchar(20));
create table invoice(invoice_number int primary key, invoice_date DATE, invoice_amount int, vehicle_no varchar(10), cust_id int, emp_id int);
ALTER TABLE `invoice` ADD FOREIGN KEY (`cust_id`) REFERENCES `customer`(`cust_id`) ON DELETE RESTRICT ON UPDATE RESTRICT; ALTER TABLE `invoice` ADD FOREIGN KEY (`emp_id`) REFERENCES `employee`(`emp_id`) ON DELETE RESTRICT ON UPDATE RESTRICT; ALTER TABLE `invoice` ADD FOREIGN KEY (`vehicle_no`) REFERENCES `vehicle`(`vehicle_number`) ON DELETE RESTRICT ON UPDATE RESTRICT;
ALTER TABLE `customer` ADD FOREIGN KEY (`cust_vehicle_no`) REFERENCES `vehicle`(`vehicle_number`) ON DELETE RESTRICT ON UPDATE RESTRICT;
alter table vehicle add customer_id int;
